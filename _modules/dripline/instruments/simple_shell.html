[docs]    def on_get(self):
        result = None
        res_lines = self._provider.send(self._on_get).split('\n')
        for line in res_lines:
            if line.startswith('Core {}:'.format(self._core)):
                result = line.split()[2].replace('\xc2\xb0', ' ')
        return result

        else:
